∀(x) | 0<=x<9000 {
    mean_[x] = Σ(data_[k][x]/10000) | 0<=k<10000
}

∀(i) | 0<=i<9000 {
    ∀(j) | 0<=j<9000 {
        cov_[i][j] = Σ(((data_[k][i] - mean_[i])*(data_[k][j] - mean_[j]))/99) | 0<=k<10000
    }
}

∀(t) | 0<=t<9000 {
    std_dev_[t] = Σ(((data_[k][t] - mean_[t])*(data_[k][t] - mean_[t]))/99) | 0<=k<10000
}

∀(p) | 0<=p<9000 {
    std_dev_[p] = √(std_dev_[p])
}

∀(i) | 0<=i<9000 {
    ∀(j) | 0<=j<9000 {
        corr[i][j] = cov_[i][j]/(std_dev_[i]*std_dev_[j])
    }
}
