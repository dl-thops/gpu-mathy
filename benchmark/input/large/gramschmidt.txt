∀(k) | 0<=k<1200 {
    norm_ = Σ(A[i][k] * A[i][k]) | 0<=i<1000
    R[k][k] =  √(norm_)
    ∀(i) | 0<=i<1000 {
        Q[i][k] = A[i][k] / R[k][k]
    }
    ∀(j) | k+1<=j<1200 {
        R[k][j] = Σ(Q[i][k] * A[i][j]) | 0<=i<1000
        ∀(i) | 0<=i<1000 {
            A[i][j] = A[i][j] - Q[i][k] * R[k][j]
        }
    }
}
